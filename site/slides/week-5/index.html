    MLE¶

    ::: {.columns align=center}

    ::: {.column width="100%"} $$ \begin{aligned} L(w) &= \prod \limits_{i = 1}^{m} P(y = y_i\ |\ x_i)\\ &= \prod \limits_{i = 1}^{m} \left[ \sigma(w^Tx_i) \right]^{y_i} \left[ 1 - \sigma(w^Tx_i) \right]^{1 - y_i} \end{aligned} $$

    MLE¶

    ::: {.columns align=center}

    ::: {.column width="100%"} $$ \begin{aligned} l(w) &= \log (L(w))\quad \quad \quad \quad \quad \quad \quad \quad \quad \quad \quad \quad \quad \quad \quad \quad \\ &= \log \left (\prod \limits_{i = 1}^{m} \left[ \sigma(w^Tx_i) \right]^{y_i} \left[ 1 - \sigma(w^Tx_i) \right]^{1 - y_i} \right) \end{aligned} $$

    :::

    MLE¶

    ::: {.columns align=center}

    ::: {.column width="100%"} $$ \begin{aligned} l(w) &= \log (L(w))\\ &= \log \left (\prod \limits_{i = 1}^{m} \left[ \sigma(w^Tx_i) \right]^{y_i} \left[ 1 - \sigma(w^Tx_i) \right]^{1 - y_i} \right)\\ &= \sum \limits_{i = 1}^{m} y_i \cdot \log (\sigma(w^T x_i)) + (1 - y_i) \log (1 - \sigma(w^Tx_i)) \end{aligned} $$
